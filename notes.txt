CREATE TABLE IF NOT EXISTS jerseys (
  id INTEGER PRIMARY KEY GENERATED ALWAYS AS IDENTITY,
  name VARCHAR(255)
  price DECIMAL(12,2)
  description VARCHAR(255)
);
